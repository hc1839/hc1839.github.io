156 template <
157 typename InputIt,
158 typename std::enable_if<
159 std::is_convertible<
160 typename std::iterator_traits<InputIt>::value_type,
161 Mat<T>>::value,
162 bool>::type = true>
163 ProjectionFactors(
164 MPI_Comm mpi_comm,
165 InputIt ortho_unitx_mos_first,
166 InputIt ortho_unitx_mos_last) {
167 if (ortho_unitx_mos_first == ortho_unitx_mos_last) {
168 throw std::logic_error(
169 "Range of orthonormal unit molecular orbitals is empty.");
170 }
171
172 // Identity matrix of the matching size.
173 const Mat<T> eye(
174 static_cast<const Mat<T> &>(*ortho_unitx_mos_first).n_rows,
175 static_cast<const Mat<T> &>(*ortho_unitx_mos_first).n_rows,
176 arma::fill::eye);
177
178 // Create projection operators for each excluded unit.
179 for (auto it = ortho_unitx_mos_first; it != ortho_unitx_mos_last; ++it) {
180 const Mat<T> &ortho_unitx_mos = *it;
181
182 const auto projx_op = MatrixProduct(
183 mpi_comm, ortho_unitx_mos, Mat<T>(ortho_unitx_mos.t()));
184
185 this->proj_ops_.push_back(eye - projx_op);
186 }
187 }

242 typename OutputIt3,
243 typename std::enable_if<
244 std::is_convertible<
245 typename std::iterator_traits<InputIt>::value_type,

256 Col<real_t>>::value,
257 bool>::type,
258 typename std::enable_if<
259 std::is_convertible<
260 typename std::iterator_traits<OutputIt3>::value_type,
261 Mat<T>>::value,
262 bool>::type>
263void SolveScfMi(
264 MPI_Comm mpi_comm,
265 const LpProjectionFactors<T> &proj_factors,
266 const Mat<T> &sys_h_op,
267 InputIt unit_basis_first,
268 OutputIt1 d_eff_h_mat_first,
269 OutputIt2 d_mo_energies_first,
270 OutputIt3 d_mos_first,
271 EigSolver<T> eig_solver) {
272 const auto &proj_ops = proj_factors.proj_ops_;
273
274 if (proj_ops.empty()) {
275 throw std::logic_error("Projection factors are empty.");
276 }
277
278 const size_t num_units = proj_ops.size();
279
280 // Input iterators.
281 auto proj_op_it = proj_ops.begin();
282 auto unit_basis_it = unit_basis_first;
283
284 // Output iterators.
285 auto d_eff_h_mat_it = d_eff_h_mat_first;
286 auto d_mo_energies_it = d_mo_energies_first;
287 auto d_mos_it = d_mos_first;
288
289 for (size_t unit_idx = 0; unit_idx != num_units; ++unit_idx) {
290 const auto &proj_op = *proj_op_it++;
291 const Mat<T> &unit_basis = *unit_basis_it++;
292
293 const Mat<T> unit_basis_t(unit_basis.t());
294
295 // Effective Hamiltonian operator.
296 const auto eff_h_op = MatrixProduct(mpi_comm, proj_op, sys_h_op, proj_op);
297
298 // Effective Hamiltonian matrix with respect to projected unit basis set.
299 const auto eff_h_mat = MatrixProduct(
300 mpi_comm, unit_basis_t, eff_h_op, unit_basis);
301
302 // Overlap matrix of projected unit basis set.
303 const auto proj_unit_basis_overlap = MatrixProduct(
304 mpi_comm, unit_basis_t, proj_op, unit_basis);
305
306 // Energies (as eigenvalues) and coefficient matrix (as eigenvectors).
307 Col<real_t> unit_mo_energies;
308 Mat<T> unit_mo_coeffs;
309
310 eig_solver(
311 unit_mo_energies, unit_mo_coeffs,
312 eff_h_mat, proj_unit_basis_overlap);
313
314 if (!unit_mo_coeffs.is_square()) {
315 throw std::runtime_error("Eigenvectors are not in a square matrix.");
316 }
317
318 if (unit_mo_coeffs.n_cols != unit_mo_energies.n_elem) {
319 throw std::runtime_error(
320 "Number of eigenvectors is not equal to the number of eigenvalues.");
321 }
322
323 // Create and output effective Hamiltonian matrix with respect to original
324 // unit basis set.
325 {
326 // Overlap matrix of original unit basis set.
327 const auto unit_basis_overlap = MatrixProduct(
328 mpi_comm, unit_basis_t, unit_basis);
329
330 const auto proj_unit_mo_coeffs = EquationSystemSolution(
331 mpi_comm,
332 unit_basis_overlap,
333 MatrixProduct(
334 mpi_comm, proj_unit_basis_overlap, unit_mo_coeffs));
335
336 // Coefficient matrix of equation system.
337 const Mat<T> eqsys_coeffs(proj_unit_mo_coeffs.t());
338
339 // Constant matrix of equation system.
340 const Mat<T> eqsys_constants(
341 MatrixProduct(
342 mpi_comm,
343 unit_basis_overlap,
344 DuoProduct(
345 mpi_comm,
346 proj_unit_mo_coeffs, unit_mo_energies.begin())).t());
347
348 static_cast<Mat<T> &>(*d_eff_h_mat_it++) = EquationSystemSolution(
349 mpi_comm, eqsys_coeffs, eqsys_constants).t();
350 }
351
352 // Indices of energies in ascending order.
353 uvec energy_sort_idxs(arma::size(unit_mo_energies));
354
355 // Output energies as real numbers in ascending order.
356 {
357 StableIndexSort(
358 unit_mo_energies.begin(),
359 unit_mo_energies.end(),
360 energy_sort_idxs.begin());
361
362 static_cast<Col<real_t> &>(*d_mo_energies_it++) =
363 unit_mo_energies.rows(energy_sort_idxs);
364 }
365
366 // Output ket matrix of molecular orbitals in ascending order by energy.
367 {
368 // Ket matrix of normalized molecular orbitals.
369 const Mat<T> unit_mos(
370 arma::normalise(
371 MatrixProduct(mpi_comm, unit_basis, unit_mo_coeffs)));
372
373 static_cast<Mat<T> &>(*d_mos_it++) = unit_mos.cols(energy_sort_idxs);
374 }
375 }
376}

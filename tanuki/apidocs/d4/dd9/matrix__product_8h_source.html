91 // Matrix A with each column multiplied by the corresponding diagonal
92 // element.
93 Mat<T> retval(arma::size(a));
94
95 // Chunks of column indices grouped by OpenMP thread of this MPI process.
96 const auto chunks = GroupIndices(0, a.n_cols, omp_get_max_threads());
97
98 #pragma omp parallel default(shared)
99 {
100 auto b_it = b_first;
101 std::advance(b_it, chunks[omp_get_thread_num()]);
102
103 for (size_t j = chunks[omp_get_thread_num()];
104 j != chunks[omp_get_thread_num() + 1];
105 ++j) {
106 retval.col(j) = a.col(j) * *b_it++;
107 }
108 }
109
110 return retval;
111}

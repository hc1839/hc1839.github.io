290 void *memptr(const ShapeType &idxs) const {
291 size_type dst_idx;
292
293 switch (storage_order()) {
294 case StorageOrder::ROW_MAJOR: {
295 size_type curr_stride = 1;
296 dst_idx = idxs.back();
297
298 for (size_type i = idxs.size() - 2; i != -1; --i) {
299 curr_stride *= shape()[i + 1];
300 dst_idx += curr_stride * idxs[i];
301 }
302
303 break;
304 }
305 case StorageOrder::COL_MAJOR: {
306 size_type curr_stride = 1;
307 dst_idx = idxs.front();
308
309 for (size_type i = 1; i != idxs.size(); ++i) {
310 curr_stride *= shape()[i - 1];
311 dst_idx += curr_stride * idxs[i];
312 }
313
314 break;
315 }
316 default: {
317 throw std::logic_error("Enumeration matching is not exhaustive.");
318 break;
319 }
320 }
321
322 return static_cast<char *>(seq().begin) + seq().item_size * dst_idx;
323 }

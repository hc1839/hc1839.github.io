163 template <
164 typename InputIt,
165 typename std::enable_if<
166 std::is_convertible<
167 typename std::iterator_traits<InputIt>::value_type,
168 Mat<T>>::value,
169 bool>::type = true>
170 ProjectionFactors(
171 MPI_Comm mpi_comm,
172 InputIt nonortho_unit_mos_first,
173 InputIt nonortho_unit_mos_last,
174 const Mat<T> &ortho_mos) {
175 if (nonortho_unit_mos_first == nonortho_unit_mos_last) {
176 throw std::logic_error(
177 "Range of nonorthogonal unit molecular orbitals is empty.");
178 }
179
180 // Projection operator.
181 const auto proj_op = MatrixProduct(
182 mpi_comm, ortho_mos, Mat<T>(ortho_mos.t()));
183
184 // Identity matrix minus projection operator.
185 const Mat<T> eye_minus_proj(
186 Mat<T>(arma::size(proj_op), arma::fill::eye) - proj_op);
187
188 uword col_idx_offset = 0;
189
190 // Create prefactor and postfactor for each unit.
191 for (auto it = nonortho_unit_mos_first;
192 it != nonortho_unit_mos_last;
193 ++it) {
194 const Mat<T> &nonortho_unit_mos = *it;
195
196 const Mat<T> ortho_unit_mos(
197 ortho_mos.cols(
198 col_idx_offset,
199 col_idx_offset + nonortho_unit_mos.n_cols - 1));
200
201 const auto partial_star_op = MatrixProduct(
202 mpi_comm, ortho_unit_mos, Mat<T>(nonortho_unit_mos.t()));
203
204 if (!partial_star_op.is_square()) {
205 throw std::logic_error("Basis dimensionality is not the same.");
206 }
207
208 this->prefactors_.push_back(eye_minus_proj + partial_star_op.t());
209 this->postfactors_.push_back(eye_minus_proj + partial_star_op);
210
211 col_idx_offset += nonortho_unit_mos.n_cols;
212 }
213 }

308void SolveScfMi(
309 MPI_Comm mpi_comm,
310 const LmoProjectionFactors<T> &proj_factors,
311 const Mat<T> &sys_h_op,
312 InputIt unit_basis_first,
313 OutputIt1 d_eff_h_mat_first,
314 OutputIt2 d_mo_energies_first,
315 OutputIt3 d_mos_first,
316 EigSolver<T> eig_solver) {
317 const auto &prefactors = proj_factors.prefactors_;
318 const auto &postfactors = proj_factors.postfactors_;
319
320 if (prefactors.empty()) {
321 throw std::logic_error("Projection factors are empty.");
322 }
323
324 if (postfactors.size() != prefactors.size()) {
325 throw std::logic_error("Mismatch in the number of factors.");
326 }
327
328 const size_t num_units = prefactors.size();
329
330 // Input iterators.
331 auto prefactor_it = prefactors.begin();
332 auto postfactor_it = postfactors.begin();
333 auto unit_basis_it = unit_basis_first;
334
335 // Output iterators.
336 auto d_eff_h_mat_it = d_eff_h_mat_first;
337 auto d_mo_energies_it = d_mo_energies_first;
338 auto d_mos_it = d_mos_first;
339
340 for (size_t unit_idx = 0; unit_idx != num_units; ++unit_idx) {
341 const auto &prefactor = *prefactor_it++;
342 const auto &postfactor = *postfactor_it++;
343 const Mat<T> &unit_basis = *unit_basis_it++;
344
345 const Mat<T> unit_basis_t(unit_basis.t());
346
347 // Effective Hamiltonian operator.
348 const auto eff_h_op = MatrixProduct(
349 mpi_comm, prefactor, sys_h_op, postfactor);
350
351 // Effective Hamiltonian matrix.
352 auto eff_h_mat = MatrixProduct(
353 mpi_comm, unit_basis_t, eff_h_op, unit_basis);
354
355 // Energies (as eigenvalues) and coefficient matrix (as eigenvectors).
356 Col<real_t> unit_mo_energies;
357 Mat<T> unit_mo_coeffs;
358
359 eig_solver(
360 unit_mo_energies, unit_mo_coeffs,
361 eff_h_mat, MatrixProduct(mpi_comm, unit_basis_t, unit_basis));
362
363 if (!unit_mo_coeffs.is_square()) {
364 throw std::runtime_error("Eigenvectors are not in a square matrix.");
365 }
366
367 if (unit_mo_coeffs.n_cols != unit_mo_energies.n_elem) {
368 throw std::runtime_error(
369 "Number of eigenvectors is not equal to the number of eigenvalues.");
370 }
371
372 // Output effective Hamiltonian matrix.
373 static_cast<Mat<T> &>(*d_eff_h_mat_it++) = std::move(eff_h_mat);
374
375 // Indices of energies in ascending order.
376 uvec energy_sort_idxs(arma::size(unit_mo_energies));
377
378 // Output energies as real numbers in ascending order.
379 {
380 StableIndexSort(
381 unit_mo_energies.begin(),
382 unit_mo_energies.end(),
383 energy_sort_idxs.begin());
384
385 static_cast<Col<real_t> &>(*d_mo_energies_it++) =
386 unit_mo_energies.rows(energy_sort_idxs);
387 }
388
389 // Output ket matrix of molecular orbitals in ascending order by energy.
390 {
391 // Ket matrix of normalized molecular orbitals.
392 const Mat<T> unit_mos(
393 arma::normalise(
394 MatrixProduct(mpi_comm, unit_basis, unit_mo_coeffs)));
395
396 static_cast<Mat<T> &>(*d_mos_it++) = unit_mos.cols(energy_sort_idxs);
397 }
398 }

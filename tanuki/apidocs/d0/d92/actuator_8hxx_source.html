106template <typename T>
107Result<T> Actuator<T>::Actuate(const Mat<T> &input, InquiryFn inquiry_fn) {
108 assert(!omp_in_parallel());
109
110 if (input.n_rows == 0 || input.n_cols < 2) {
111 throw std::length_error("Invalid size of the input matrix to transform.");
112 }
113
114 if (sidedness_ == JacobiSidedness::TWO_SIDED && !input.is_square()) {
115 throw std::invalid_argument(
116 "GRS is two-sided but the input matrix is not square.");
117 }
118
119 // Column indices as competitors in a round-robin tournament, where each
120 // round is a non-conflicting rotation set.
121 RoundRobinTourney<long long> tourney(input.n_cols);
122
123 // Number of rotation sets for each group.
124 const auto group_sizes = GroupSizes(0, tourney.num_rounds(), num_groups_);
125
126 // Shared memory for the previous matrix.
127 MpiSharedMemory prev_mat_shm(
128 mpi_comm_, shared_mem_prev_mat_name_,
129 sizeof(T) * input.n_elem, open_or_create);
130
131 // Pointer to the shared memory for the previous matrix in the iteration.
132 const auto prev_mat_shm_ptr = static_cast<T *>(prev_mat_shm.mem_address());
133
134 parallel::memory::Copy(
135 host_based_comms_.intrahost(),
136 input.memptr(),
137 prev_mat_shm_ptr,
138 sizeof(T) * input.n_elem);
139
140 // Matrix from the previous iteration to check for convergence.
141 Mat<T> prev_matrix(
142 prev_mat_shm_ptr,
143 input.n_rows, input.n_cols,
144 false, true);
145
146 // Shared memory for the current matrix.
147 MpiSharedMemory curr_mat_shm(
148 mpi_comm_, shared_mem_curr_mat_name_,
149 sizeof(T) * input.n_elem, open_or_create);
150
151 // Pointer to the shared memory for the current matrix in the iteration.
152 const auto curr_mat_shm_ptr = static_cast<T *>(curr_mat_shm.mem_address());
153
154 parallel::memory::Copy(
155 host_based_comms_.intrahost(),
156 input.memptr(),
157 curr_mat_shm_ptr,
158 sizeof(T) * input.n_elem);
159
160 // Matrix in the current iteration.
161 Mat<T> curr_matrix(
162 curr_mat_shm_ptr,
163 input.n_rows, input.n_cols,
164 false, true);
165
166 real_t relaxation = init_relax_;
167
168 Result<T> retval;
169 retval.has_converged = false;
170
171 for (size_t iter = 0; iter != max_iterations_; ++iter) {
172 // Iterator to a rotation set.
173 auto rs_it = tourney.begin();
174
175 // Iterate over groups of rotation sets.
176 for (size_t gr = 0; gr != num_groups_; ++gr) {
177 // Rotation pairs, corresponding by column, in the group.
178 Mat<long long> rotation_pairs;
179
180 // Concatenate rotation sets in the group.
181 for (size_t rs = 0; rs != group_sizes[gr]; ++rs) {
182 rotation_pairs = arma::join_horiz(rotation_pairs, rs_it->t());
183 ++rs_it;
184 }
185
186 // Matrix of relaxed rotation cosines and sines at the top and bottom
187 // rows, respectively, for rotation pairs.
188 Mat<real_t> cosine_sine(2, rotation_pairs.n_cols);
189
190 // Delimitation of rotation pairs by MPI processes.
191 const auto batch_indices = GroupIndices(
192 0, rotation_pairs.n_cols, mpi_comm_size_);
193
194 // Delimitation of rotation pairs by threads.
195 const auto chunk_indices = GroupIndices(
196 batch_indices[mpi_rank_],
197 batch_indices[mpi_rank_ + 1],
198 max_threads());
199
200 #pragma omp parallel default(shared)
201 {
202 const auto thread_num = omp_get_thread_num();
203
204 if (thread_num < max_threads()) {
205 // Calculate and store the relaxed rotation for each rotation pair.
206 for (size_t rp = chunk_indices[thread_num];
207 rp != chunk_indices[thread_num + 1];
208 ++rp) {
209 MatrixIndexPair indices = {
210 .type = MatrixIndexPair::PairType::COLUMNS,
211 .first = static_cast<size_t>(rotation_pairs(0, rp)),
212 .second = static_cast<size_t>(rotation_pairs(1, rp))
213 };
214
215 // Ensure the first index is not greater than the second.
216 if (indices.first > indices.second) {
217 std::swap(indices.first, indices.second);
218 }
219
220 if (indices.first == -1 || indices.second == -1) {
221 cosine_sine(0, rp) = 1.0;
222 cosine_sine(1, rp) = 0.0;
223 } else {
224 const IndexedVectorPair<T> vectors = {
225 .indices = indices,
226 .first = Col<T>(
227 curr_matrix.colptr(indices.first),
228 curr_matrix.n_rows,
229 false,
230 true),
231 .second = Col<T>(
232 curr_matrix.colptr(indices.second),
233 curr_matrix.n_rows,
234 false,
235 true)
236 };
237
238 // Unrelaxed rotation.
239 const auto rotation_spec = inquiry_fn(vectors);
240
241 const auto relaxed_angle = (1.0 - relaxation) *
242 std::atan2(rotation_spec.sine, rotation_spec.cosine);
243
244 cosine_sine(0, rp) = std::cos(relaxed_angle);
245 cosine_sine(1, rp) = std::sin(relaxed_angle);
246 }
247 }
248 }
249 }
250
251 // Broadcast the relaxed rotations.
252 for (int rank = 0; rank != mpi_comm_size_; ++rank) {
253 // Number of elements to broadcast.
254 const size_t num_elements =
255 (batch_indices[rank + 1] - batch_indices[rank]) * 2;
256
257 MPI_Bcast(
258 cosine_sine.colptr(batch_indices[rank]),
259 num_elements,
260 MPI_DOUBLE,
261 rank,
262 mpi_comm_);
263 }
264
265 const auto num_pairs_per_round = tourney.num_pairs_per_round(true);
266
267 // Apply the rotations in each rotation set in a distributed manner.
268 for (size_t start_rp = 0;
269 start_rp != rotation_pairs.n_cols;
270 start_rp += num_pairs_per_round) {
271
272 const Mat<long long> rotation_set(
273 rotation_pairs.colptr(start_rp),
274 2,
275 num_pairs_per_round,
276 false,
277 true);
278
279 vector<RotationMatrixSpec> rotation_specs;
280
281 // Populate the rotation specs.
282 for (size_t rp = start_rp;
283 rp != start_rp + num_pairs_per_round;
284 ++rp) {
285 RotationMatrixSpec rotation_spec = {
286 .cosine = cosine_sine(0, rp),
287 .sine = cosine_sine(1, rp)
288 };
289
290 rotation_specs.push_back(rotation_spec);
291 }
292
293 DistApplyRotationSet(rotation_set, rotation_specs, true, curr_matrix);
294
295 if (sidedness_ == JacobiSidedness::TWO_SIDED) {
296 DistApplyRotationSet(
297 rotation_set, rotation_specs, false, curr_matrix);
298 }
299 }
300
301 relaxation = relax_fn_(iter, relaxation, gr);
302
303 if (relaxation < 0.0 || relaxation >= 1.0) {
304 throw std::domain_error(
305 "Relaxation parameter is not in the interval [0, 1).");
306 }
307 }
308
309 auto has_converged = convergence_checker_(prev_matrix, curr_matrix);
310
311 MPI_Allreduce(
312 MPI_IN_PLACE, &has_converged, 1, MPI_CXX_BOOL, MPI_LAND, mpi_comm_);
313
314 retval.num_iters = iter + 1;
315
316 if (has_converged) {
317 retval.has_converged = true;
318 break;
319 } else {
320 parallel::memory::Copy(
321 host_based_comms_.intrahost(),
322 curr_matrix.memptr(),
323 prev_matrix.memptr(),
324 curr_matrix.n_elem * sizeof(T));
325 }
326 }
327
328 MPI_Barrier(mpi_comm_);
329
330 retval.transform = Mat<T>(
331 curr_matrix.memptr(),
332 curr_matrix.n_rows,
333 curr_matrix.n_cols);
334
335 return retval;
336}

348template <typename T>
349void Actuator<T>::DistApplyRotationSet(
350 const arma::Mat<long long> &rotation_set,
351 const std::vector<RotationMatrixSpec> &rotation_specs,
352 bool is_by_col,
353 Mat<T> &matrix) const {
354 assert(!omp_in_parallel());
355 assert(!rotation_set.is_empty());
356 assert(rotation_set.n_rows == 2);
357 assert(rotation_specs.size() == rotation_set.n_cols);
358
359 if (!is_by_col) {
360 Mat<T> matrix_st(matrix.st());
361 DistApplyRotationSet(rotation_set, rotation_specs, true, matrix_st);
362 matrix = matrix_st.st();
363 return;
364 }
365
366 int intrahost_rank;
367 MPI_Comm_rank(host_based_comms_.intrahost(), &intrahost_rank);
368
369 int intrahost_size;
370 MPI_Comm_size(host_based_comms_.intrahost(), &intrahost_size);
371
372 // Subdivide the rotation set by MPI process.
373 const auto rp_batches = GroupIndices(
374 0, rotation_specs.size(), intrahost_size);
375
376 // Subdivide the batch of rotation pairs by thread.
377 const auto rp_chunks = GroupIndices(
378 rp_batches[intrahost_rank],
379 rp_batches[intrahost_rank + 1],
380 max_threads());
381
382 // Concurrently rotate the columns assigned to this MPI process.
383 #pragma omp parallel default(shared)
384 {
385 const auto thread_num = omp_get_thread_num();
386
387 // Number of rotations assigned to this thread.
388 const size_t num_rotations = thread_num < max_threads() ?
389 rp_chunks[thread_num + 1] - rp_chunks[thread_num] : 0;
390
391 if (num_rotations != 0) {
392 // Iterator to the beginning of the chunk of rotation specs.
393 auto spec_first = rotation_specs.begin();
394 std::advance(spec_first, rp_chunks[thread_num]);
395
396 // Iterator to the end of the chunk of rotation specs for loop
397 // termination.
398 auto spec_last = spec_first;
399
400 std::advance(spec_last, num_rotations);
401
402 // Iterator to the current rotation pair.
403 auto pair_it = rotation_set.begin_col(rp_chunks[thread_num]);
404
405 // Apply the rotations assigned to this thread.
406 for (auto spec_it = spec_first;
407 spec_it != spec_last;
408 ++spec_it, std::advance(pair_it, 2)) {
409 pair<long long, long long> rotation_pair(*pair_it, *(pair_it + 1));
410
411 if (rotation_pair.first > rotation_pair.second) {
412 std::swap(rotation_pair.first, rotation_pair.second);
413 }
414
415 if (rotation_pair.first == -1) {
416 continue;
417 }
418
419 const auto &rotation_spec = *spec_it;
420
421 const auto row_rotation =
422 CreateGivensRotation(rotation_spec, 2, 0, 1);
423
424 const uvec idxs = {
425 static_cast<size_t>(rotation_pair.first),
426 static_cast<size_t>(rotation_pair.second)
427 };
428
429 matrix.cols(idxs) = matrix.cols(idxs) * row_rotation.t();
430 }
431 }
432 }
433
434 // Wait for the MPI processes at this host to finish applying the rotations.
435 MPI_Barrier(host_based_comms_.intrahost());
436}

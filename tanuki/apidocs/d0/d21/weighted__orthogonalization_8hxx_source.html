53 template <typename T>
54 RotationMatrixSpec rotate(
55 const Col<T> &norm_col1,
56 const Col<T> &ortho_col1,
57 real_t weight1,
58 const Col<T> &norm_col2,
59 const Col<T> &ortho_col2,
60 real_t weight2,
61 real_t zero_abs_thresh) {
62 assert(
63 norm_col1.n_rows == norm_col2.n_rows &&
64 norm_col1.n_rows == ortho_col1.n_rows &&
65 norm_col2.n_rows == ortho_col2.n_rows);
66 assert(weight1 >= 0.0);
67 assert(weight2 >= 0.0);
68
69 // Do not rotate if both weights are close enough to zero.
70 if (weight1 < zero_abs_thresh && weight2 < zero_abs_thresh) {
71 return CreateIdentityRotation();
72 }
73
74 // Intermediate values for the B coefficient.
75 const auto B1 = weight1 * as_scalar(norm_col1.t() * ortho_col1);
76 const auto B2 = weight2 * as_scalar(norm_col2.t() * ortho_col2);
77
78 // B coefficient.
79 const complex_t B = B1 + B2;
80
81 // Intermediate values for the C coefficient.
82 const auto C1 = weight2 * as_scalar(norm_col2.t() * ortho_col1);
83 const auto C2 = -weight1 * as_scalar(norm_col1.t() * ortho_col2);
84
85 // C coefficient.
86 const complex_t C = C1 + C2;
87
88 // A coefficient.
89 const auto A = sqrt(B * B + C * C);
90
91 // Do not rotate if the A coefficient is close enough to zero.
92 if (abs(A) < zero_abs_thresh) {
93 return CreateIdentityRotation();
94 }
95
96 return {
97 .cosine = (B / A).real(),
98 .sine = (C / A).real()
99 };
100 }

105parallel::grs::Result<T> WeightOrthogonalized(
106 const Mat<T> &nonortho_matrix,
107 const Mat<T> &prelim_ortho_matrix,
108 const vector<real_t> &weights,
109 real_t zero_abs_thresh,
110 Actuator<T> &actuator) {
111 assert(!nonortho_matrix.is_empty());
112 assert(arma::size(prelim_ortho_matrix) == arma::size(nonortho_matrix));
113 assert(weights.size() == nonortho_matrix.n_cols);
114 assert(zero_abs_thresh > 0.0);
115
116 // Ensure all weights are non-negative.
117 assert(
118 all_of(
119 weights.begin(), weights.end(),
120 [](const real_t &weight) -> bool { return weight >= 0.0; }));
121
122 // Inquiry function for the GRS actuator.
123 auto inquiry_fn = [&nonortho_matrix, &weights, zero_abs_thresh](
124 const IndexedVectorPair<T> &vectors) -> RotationMatrixSpec {
125 const auto &first_idx = vectors.indices.first;
126 const auto &second_idx = vectors.indices.second;
127
128 return rotate(
129 Col<T>(
130 const_cast<T *>(nonortho_matrix.colptr(first_idx)),
131 nonortho_matrix.n_rows, false, true),
132 vectors.first,
133 weights[first_idx],
134 Col<T>(
135 const_cast<T *>(nonortho_matrix.colptr(second_idx)),
136 nonortho_matrix.n_rows, false, true),
137 vectors.second,
138 weights[second_idx],
139 zero_abs_thresh);
140 };
141
142 return actuator.Actuate(prelim_ortho_matrix, inquiry_fn);
143}

145template <typename T>
146parallel::grs::Result<T> WeightOrthogonalized(
147 MPI_Comm mpi_comm,
148 const Mat<T> &nonortho_matrix,
149 const Mat<T> &prelim_ortho_matrix,
150 const vector<real_t> &weights,
151 size_t max_sweeps,
152 real_t zero_abs_thresh) {
153 assert(max_sweeps > 0);
154
155 int mpi_comm_size;
156 MPI_Comm_size(mpi_comm, &mpi_comm_size);
157
158 // Function that determines whether GRS has converged.
159 auto convergence_checker =
160 [zero_abs_thresh](const Mat<T> &prev, const Mat<T> &curr) -> bool {
161 return norm(curr - prev, "fro") < zero_abs_thresh;
162 };
163
164 // Initial relaxation parameter for GRS.
165 const auto init_relax_param =
166 GrsOneSidedRelaxParam(prelim_ortho_matrix.n_cols);
167
168 // Function that halves the relaxation parameter after sweeping each group of
169 // rotation sets.
170 auto relax_fn = [](size_t, real_t prev_relax, size_t) -> real_t {
171 return prev_relax * 0.5;
172 };
173
174 // Create the default GRS actuator for WO.
175 parallel::grs::Actuator<T> actuator(
176 mpi_comm,
177 omp_get_max_threads(),
178 JacobiSidedness::ONE_SIDED_RIGHT,
179 init_relax_param,
180 mpi_comm_size,
181 max_sweeps,
182 relax_fn,
183 convergence_checker);
184
185 return WeightOrthogonalized(
186 nonortho_matrix,
187 prelim_ortho_matrix,
188 weights,
189 zero_abs_thresh,
190 actuator);
191}

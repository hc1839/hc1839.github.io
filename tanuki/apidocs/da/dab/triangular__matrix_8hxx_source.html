48template <typename T>
49void ForwardSubstitute(
50 const Mat<T> &lower_coeffs,
51 const Mat<T> &constants,
52 size_t start_col,
53 size_t end_col_exclusive,
54 Mat<T> &solutions) {
55 assert(start_col >= 0);
56 assert(start_col <= end_col_exclusive);
57 assert(end_col_exclusive <= constants.n_cols);
58 assert(arma::size(solutions) == arma::size(constants));
59
60 if (end_col_exclusive == start_col) {
61 return;
62 }
63
64 // Initialize the values that will become the solutions.
65 solutions.submat(0, start_col, solutions.n_rows - 1, end_col_exclusive - 1) =
66 constants.submat(
67 0, start_col, constants.n_rows - 1, end_col_exclusive - 1);
68
69 // Loop over the columns of the constants on the right-hand side.
70 for (size_t rhs_col = start_col; rhs_col != end_col_exclusive; ++rhs_col) {
71 // Calculate each solution of the column.
72 for (size_t curr_sol_idx = 0;
73 curr_sol_idx != solutions.n_rows;
74 ++curr_sol_idx) {
75 const auto solution = solutions(curr_sol_idx, rhs_col) /
76 lower_coeffs(curr_sol_idx, curr_sol_idx);
77 solutions.col(rhs_col) = solutions.col(rhs_col) -
78 lower_coeffs.col(curr_sol_idx) * solution;
79 solutions(curr_sol_idx, rhs_col) = solution;
80 }
81 }
82}

141template <typename T>
142Mat<T> ForwardSubstitute(
143 MPI_Comm mpi_comm, const Mat<T> &lower_coeffs, const Mat<T> &constants) {
144 int intrahost_rank;
145 int intrahost_size;
146
147 // Allocate shared memory.
148 unique_ptr<MpiSharedMemory> shm(
149 new MpiSharedMemory(
150 mpi_comm,
151 "triangular_matrix_substitution_cache",
152 sizeof(T) * constants.n_elem,
153 open_or_create));
154
155 MPI_Comm_rank(shm->comms().intrahost(), &intrahost_rank);
156 MPI_Comm_size(shm->comms().intrahost(), &intrahost_size);
157
158 const int intrahost_color = shm->comms().intrahost_color();
159 const size_t num_hosts = shm->comms().hosts().num_hosts();
160
161 unique_ptr<Mat<T>> solutions(
162 new Mat<T>(
163 static_cast<T *>(shm->mem_address()),
164 constants.n_rows, constants.n_cols,
165 false, true));
166
167 // Indices of the batches grouped by host.
168 const auto host_batches = GroupIndices(0, constants.n_cols, num_hosts);
169
170 // Indices of the local batches grouped by MPI process at this host.
171 const auto local_batches = GroupIndices(
172 host_batches[intrahost_color],
173 host_batches[intrahost_color + 1],
174 intrahost_size);
175
176 // Indices of the chunks grouped by OpenMP thread of this MPI process.
177 const auto chunks = GroupIndices(
178 local_batches[intrahost_rank],
179 local_batches[intrahost_rank + 1],
180 omp_get_max_threads());
181
182 #pragma omp parallel default(shared)
183 {
184 ForwardSubstitute(
185 lower_coeffs,
186 constants,
187 chunks[omp_get_thread_num()],
188 chunks[omp_get_thread_num() + 1],
189 *solutions);
190 }
191
192 MPI_Barrier(mpi_comm);
193
194 if (intrahost_rank == 0) {
195 for (int host_color = 0; host_color != num_hosts; ++host_color) {
196 const size_t num_cols =
197 host_batches[host_color + 1] - host_batches[host_color];
198
199 const size_t num_elem = num_cols * solutions->n_rows;
200
201 MPI_Bcast(
202 solutions->colptr(host_batches[host_color]),
203 num_elem,
204 MpiBasicDatatype<T>(),
205 host_color,
206 shm->comms().interhost());
207 }
208 }
209
210 MPI_Barrier(mpi_comm);
211
212 return Mat<T>(*solutions);
213}
